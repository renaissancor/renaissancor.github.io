

### Copy Operator and 

```c++
class CNew
{
private:
    int m_i; 
    short m_s; 

public: 
    void SetInt(int _a)
    {
        m_i = a; 
    }
    // Default Assignment operator 
    void operator=(const CNew& _Other){
        m_i = _Other.m_i; 
        m_s = _Other.m_s; 

        return *this; 
    }
public: 
    // 湲곕낯 Default Copy operator 臾대젰?占쏙옙 
    CNew(const CNew & _Other) = delete; // 留됱븘 踰꾨━占?? 
    CNew(const CNew & _Other)
    :m_i(_Other.m_i), m_s(_Other.m_s)
    {}
}
```
